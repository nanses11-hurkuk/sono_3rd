+ public class ExchangeSessionManager {
+     private static final int    MAX_RETRY  = 3;
+     private static final long[] BACKOFF_MS = { 1_000L, 2_000L, 4_000L };
+     @Autowired private ResourceManager resourceManager;
+
+     // EWS 세션 획득 (재시도 포함)
+     public ExchangeService getEwsSession(String domainId, String mailAddress)
+             throws Exception {
+         return executeWithRetry(() -> {
+             ExchangeConfigDTO config = resourceManager.get(domainId).getEWSConfigDTO();
+             ExchangeService svc = EWSCommon.getExchangeService(config, mailAddress);
+             svc.getServerInfo();  // Ping 유효성 검증
+             return svc;
+         });
+     }

+     // LDAP 초기화 (재시도 포함)
+     public void initLdapWithRetry(LdapManager ldapManager, String domainId)
+             throws Exception {
+         executeWithRetry(() -> {
+             ExchangeConfigDTO config = resourceManager.get(domainId).getEWSConfigDTO();
+             ldapManager.initLdapConfig(config);
+             return null;
+         });
+     }
+
+     // Exponential Backoff 재시도 공통 템플릿
+     private <T> T executeWithRetry(CheckedSupplier<T> action) throws Exception {
+         int attempt = 0;
+         while (attempt < MAX_RETRY) {
+             try { return action.get(); }
+             catch (Exception e) {
+                 attempt++;
+                 if (attempt >= MAX_RETRY) throw e;
+                 Thread.sleep(BACKOFF_MS[attempt - 1]);
+             }
+         }
+         throw new Exception("세션 초기화 실패 (최대 재시도 초과)");
+     }
+     @FunctionalInterface interface CheckedSupplier<T> { T get() throws Exception; }
+ }
