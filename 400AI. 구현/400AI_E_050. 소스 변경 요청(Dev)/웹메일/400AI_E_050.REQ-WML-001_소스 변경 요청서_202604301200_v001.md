// [+] 대용량 첨부 업로드 클라이언트 로직 (GAP-WML-001-05)
async function uploadLargeAttachment(file, draftMessageId, mailBoxKey) {
    const CHUNK_SIZE = 5 * 1024 * 1024; // 5MB

    // 1. 업로드 세션 생성
    const { sessionId } = await POST('/uploadSession', {
        draftMessageId, fileName: file.name,
        fileSize: file.size, contentType: file.type
    });

    // 2. 청크 분할 전송
    let start = 0;
    while (start < file.size) {
        const end   = Math.min(start + CHUNK_SIZE - 1, file.size - 1);
        const chunk = file.slice(start, end + 1);

        const progress = await PUT(`/uploadSession/${sessionId}`, chunk, {
            'Content-Range': `bytes ${start}-${end}/${file.size}`
        });

        // 진행률 업데이트
        updateProgressBar(Math.round(((end + 1) / file.size) * 100));

        if (progress.completed) break;
        start = end + 1;
    }
}
