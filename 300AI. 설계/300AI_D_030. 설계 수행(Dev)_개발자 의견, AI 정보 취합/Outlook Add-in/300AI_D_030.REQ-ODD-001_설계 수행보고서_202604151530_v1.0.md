```java
// [신규] ExchangeSessionManager.java
// 위치: egovframework.covimail.common.service.ExchangeSessionManager
public class ExchangeSessionManager {
    private static final int MAX_RETRY = 3;
    private static final long[] BACKOFF_MS = {1000L, 2000L, 4000L};

    @Autowired private ResourceManager resourceManager; // coviexchange 재활용

    public ExchangeService getSession(String domainId, String mailAddress) throws Exception {
        int attempt = 0;
        while (attempt < MAX_RETRY) {
            try {
                // EWSCommon.getExchangeService() 재활용 (smarts4j_coviexchange)
                ExchangeConfigDTO config = resourceManager.get(domainId).getEWSConfigDTO();
                ExchangeService service = EWSCommon.getExchangeService(config, mailAddress);
                // 세션 유효성 검증 (간단한 Ping)
                service.getServerInfo(); // or ResolveName 호출
                return service;
            } catch (Exception e) {
                attempt++;
                if (attempt >= MAX_RETRY) throw e;
                Thread.sleep(BACKOFF_MS[attempt - 1]);
            }
        }
        throw new Exception("Exchange 세션 연결 실패 (최대 재시도 초과)");
    }
}
```
