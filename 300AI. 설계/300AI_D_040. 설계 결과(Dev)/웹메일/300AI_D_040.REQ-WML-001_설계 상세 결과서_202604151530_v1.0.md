#### [GAP-WML-001-02] AccessToken 갱신 — TokenRefreshService

```java
/**
 * TokenRefreshService.java
 * 위치: egovframework.covimail.common.service.impl.TokenRefreshService
 * 산출 근거: 300AI_D_030.REQ-WML-001, 300AI_D_020.REQ-WML-001
 *
 * 핵심 로직:
 *  1. Redis 캐시에서 토큰 조회
 *  2. 없거나 만료 임박 시 MSAL4J ClientCredentials Flow로 재발급
 *  3. Redis에 TTL 3500초로 저장
 *  4. EWSCommon에서 Bearer 토큰 인증 활성화 (주석 해제)
 */
@Service
public class TokenRefreshService {

    @Autowired private RedisDataUtil redisDataUtil;

    public String getOrRefreshToken(String domainId) throws Exception {
        String cacheKey = "msAccessToken:" + domainId;

        // Step 1: Redis 캐시 조회
        String cachedToken = redisDataUtil.get(cacheKey);
        if (cachedToken != null && !cachedToken.isEmpty()) {
            return cachedToken;
        }

        // Step 2: Redis 미존재 or 만료 → MSAL4J 재발급
        String clientId     = RedisDataUtil.getBaseConfig("OauthClientId",     domainId);
        String clientSecret = RedisDataUtil.getBaseConfig("OauthClientSecret", domainId);
        String tenantId     = RedisDataUtil.getBaseConfig("OauthTenantId",     domainId);
        String scope        = "https://outlook.office365.com/.default";

        ConfidentialClientApplication app = ConfidentialClientApplication
            .builder(clientId, ClientCredentialFactory.createFromSecret(clientSecret))
            .authority("https://login.microsoftonline.com/" + tenantId)
            .build();

        IAuthenticationResult result = app
            .acquireToken(ClientCredentialParameters.builder(
                Collections.singleton(scope)).build())
            .get(10, TimeUnit.SECONDS);  // 최대 10초 대기

        String newToken = result.accessToken();

        // Step 3: Redis 캐싱 (TTL: 3,500초 — 만료 100초 전 갱신)
        redisDataUtil.setEx(cacheKey, newToken, 3500);

        return newToken;
    }

    public void invalidateToken(String domainId) {
        redisDataUtil.delete("msAccessToken:" + domainId);
    }
}
```

```java
public String createUploadSession(String mailBoxKey, String draftMessageId,
                                   String fileName, long fileSize, String domainId) throws Exception {
    // Graph API: POST /v1.0/users/{mailBoxKey}/messages/{draftMessageId}/attachments/createUploadSession
    CoviMap body = new CoviMap();
    body.put("AttachmentItem", Map.of(
        "attachmentType", "file",
        "name",           fileName,
        "size",           fileSize
    ));

    // CoviMSAPIUtil 기반 Graph API 호출 (기존 인프라 재활용)

}

public void uploadChunk(String uploadUrl, byte[] chunk,
                         long rangeStart, long rangeEnd, long totalSize) throws Exception {
    // PUT {uploadUrl}
    // Content-Range: bytes {rangeStart}-{rangeEnd}/{totalSize}
    // Content-Length: chunk.length
    // 타임아웃: connectionTimeout=5s, readTimeout=60s (대용량 파일 고려)
    HttpPut put = new HttpPut(uploadUrl);
    put.setHeader("Content-Range",
        String.format("bytes %d-%d/%d", rangeStart, rangeEnd, totalSize));
    put.setHeader("Content-Length", String.valueOf(chunk.length));
    put.setEntity(new ByteArrayEntity(chunk));

    // CloseableHttpClient (connectionTimeout=5000, socketTimeout=60000)
    try (CloseableHttpResponse resp = httpClient.execute(put)) {
        int status = resp.getStatusLine().getStatusCode();
        if (status != 200 && status != 201 && status != 202) {
            throw new IOException("청크 업로드 실패: HTTP " + status);
        }
    }
}
```

#### [GAP-WML-001-09] 웹훅 구독 자동 갱신 스케줄러

```java
/**
 * SubscriptionRenewalScheduler.java
 * 위치: egovframework.covimail.common.util.SubscriptionRenewalScheduler
 * 산출 근거: 300AI_D_040.REQ-WML-001
 * 참조 패턴: AsyncTaskAutoDelete.java (covimail 기존 스케줄러 패턴)
 * 재활용: MailSvc.createSubscription() (smarts4j_coviexchange 기 구현)
 */
@Component
public class SubscriptionRenewalScheduler {

    @Autowired private ResourceManager resourceManager;  // coviexchange 재활용
    @Autowired private RedisDataUtil    redisDataUtil;

    private static final long RENEWAL_THRESHOLD_MINUTES = 60L; // 만료 60분 전 갱신

    @Scheduled(fixedDelay = 3_600_000) // 60분 주기 실행
    public void renewSubscriptions() {
        Set<String> keys = redisDataUtil.keys("ewsSubscription:*");
        if (keys == null || keys.isEmpty()) return;

        for (String key : keys) {
            try {
                EwsSubscriptionInfo info =
                    redisDataUtil.getObject(key, EwsSubscriptionInfo.class);

                long minutesLeft = ChronoUnit.MINUTES.between(
                    LocalDateTime.now(), info.getExpiresAt());

                if (minutesLeft <= RENEWAL_THRESHOLD_MINUTES) {
                    // EWS ExchangeService 생성 (EWSCommon 재활용)
                    ExchangeConfigDTO config =
                        resourceManager.get(info.getDomainId()).getEWSConfigDTO();
                    ExchangeService service =
                        EWSCommon.getExchangeService(config, info.getMailAddress());

                    // MailSvc.createSubscription() 재활용 (smarts4j_coviexchange)
                    MailSvcImpl mailSvc = new MailSvcImpl();
                    GWSubscribeEwsDTO dto = new GWSubscribeEwsDTO();
                    dto.setSubscriptionId(info.getSubscriptionId());
                    MSResponseListDataDTO<MSSubscribeEwsDTO> result =
                        mailSvc.createSubscription(service, dto);

                    // Redis TTL 갱신 (최대 4,230분 = 253,800초)
                    String newSubId = result.getItems().get(0).getSubscriptionId();
                    info.setSubscriptionId(newSubId);
                    info.setExpiresAt(LocalDateTime.now().plusMinutes(4230));
                    redisDataUtil.setEx(key, info, 253_800);
                }
            } catch (Exception e) {
                LOGGER.error("구독 갱신 실패 key={} : {}", key, e.getMessage(), e);
            }
        }
    }
}
```
