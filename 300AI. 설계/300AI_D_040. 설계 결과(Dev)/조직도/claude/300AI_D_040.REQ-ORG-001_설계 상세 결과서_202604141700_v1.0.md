// HRConnector JDBC 로직 (HRConnectCode.java 재활용)
public void executeSync() {
    Connection erpConn = DriverManager.getConnection(
        "jdbc:sqlserver://{host}:1433;databaseName=UTICommon", DB_USER, DB_PASS);
    List<Map> deptData = fetchRows(erpConn, "SELECT * FROM dbo.GW_Department");
    List<Map> userData = fetchRows(erpConn, "SELECT * FROM dbo.GW_User");

        orgDeptMapper.upsertDept(dept);

        orgUserMapper.upsertUser(user, jobLevelCode);
    }
