// LdapManager/LdapUserSvc 재활용 (coviexchange 기존 구현)
public void syncToAD(UserDTO user) {
    LdapContext ctx = ldapManager.getContext();
    // UPN 표준화: LogonID@domain.com
    Attributes attrs = new BasicAttributes();
    attrs.put("sAMAccountName", user.getLogonID());
    attrs.put("userPrincipalName", user.getLogonID() + "@utivina.com");
    attrs.put("displayName", user.getDisplayName());
    attrs.put("mail", user.getLogonID() + "@utivina.com");
    ctx.modifyAttributes(userDN, DirContext.REPLACE_ATTRIBUTE, attrs);
}
