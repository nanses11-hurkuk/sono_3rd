/**
 * ExchangeSessionManager.java
 * 위치: egovframework.covimail.common.service.ExchangeSessionManager
 * 핵심: LdapManager.initLdapConfig() 매 요청 재생성 → 재시도 래퍼로 안정성 확보
 * 산출 근거: 300AI_D_040.REQ-ODD-001
 */
@Component
public class ExchangeSessionManager {

    private static final int    MAX_RETRY  = 3;
    private static final long[] BACKOFF_MS = {1000L, 2000L, 4000L};

    @Autowired private ResourceManager resourceManager; // coviexchange 재활용

    /**
     * EWS ExchangeService 획득 (재시도 포함)
     */
    public ExchangeService getEwsSession(String domainId, String mailAddress)
            throws Exception {
        return executeWithRetry(() -> {
            ExchangeConfigDTO config =
                resourceManager.get(domainId).getEWSConfigDTO();
            ExchangeService svc = EWSCommon.getExchangeService(config, mailAddress);
            svc.getServerInfo(); // Ping 유효성 검증
            return svc;
        });
    }

    /**
     * LDAP 초기화 (재시도 포함)
     * LdapManager.initLdapConfig() 매 호출 재생성 구조를 감싸서 안정성 확보
     */
    public void initLdapWithRetry(LdapManager ldapManager, String domainId)
            throws Exception {
        executeWithRetry(() -> {
            ExchangeConfigDTO config =
                resourceManager.get(domainId).getEWSConfigDTO();
            ldapManager.initLdapConfig(config);
            return null;
        });
    }

    private <T> T executeWithRetry(CheckedSupplier<T> action) throws Exception {
        int attempt = 0;
        while (attempt < MAX_RETRY) {
            try {
                return action.get();
            } catch (Exception e) {
                attempt++;
                if (attempt >= MAX_RETRY) throw e;
                Thread.sleep(BACKOFF_MS[attempt - 1]);
            }
        }
        throw new Exception("세션 초기화 실패 (최대 재시도 초과)");
    }

    @FunctionalInterface
    interface CheckedSupplier<T> {
        T get() throws Exception;
    }
}

// ─── OAuth2 구현체 (신규 — GraphServiceContext.accessToken 활용) ──────
@Component("oauth2AuthStrategy")
public class OAuth2AuthStrategy implements OutlookAuthStrategy {

    @Override
    public String authenticate(String domainId, String mailAddress) throws Exception {
        String cacheKey = "outlookToken:" + domainId + ":" + mailAddress;

        // 1. Redis 캐시 조회
        String cachedToken = RedisDataUtil.get(cacheKey);
        if (cachedToken != null) return cachedToken;

        // 2. MSAL4J ClientCredentials Flow
        String clientId     = RedisDataUtil.getBaseConfig("OauthClientId",     domainId);
        String clientSecret = RedisDataUtil.getBaseConfig("OauthClientSecret", domainId);
        String tenantId     = RedisDataUtil.getBaseConfig("OauthTenantId",     domainId);

        ConfidentialClientApplication app = ConfidentialClientApplication
            .builder(clientId, ClientCredentialFactory.createFromSecret(clientSecret))
            .authority("https://login.microsoftonline.com/" + tenantId)
            .build();

        IAuthenticationResult result = app.acquireToken(
            ClientCredentialParameters.builder(
                Collections.singleton("https://outlook.office365.com/.default"))
            .build()).get(10, TimeUnit.SECONDS);

        // 3. GraphServiceContext에 설정 (smart4j_msapi 구조 활용)
        String token = result.accessToken();
        RedisDataUtil.setEx(cacheKey, token, 3500);
        return token;
    }

    @Override public String getUserType() { return "EXTERNAL_OAUTH2"; }
}

/**
 * OutlookAddinSvcImpl.manageDistGroup()

 * 산출 근거: 300AI_D_040.REQ-ODD-001

public void manageDistGroup(HttpServletRequest request,
                             HttpServletResponse response,
                             String[] arrParams) throws Exception {
    String mailAddress = arrParams[0];
    String opType      = arrParams[3]; // CREATE / MODIFY / DELETE / ADD_MEMBER / REMOVE_MEMBER
    CoviMap param      = CoviMap.fromJSON(arrParams[4]);

    // Redis Distributed Lock (G-05)
    String groupCode = param.getString("GroupCode");
    String lockKey   = "lock:outlook:distgroup:" + groupCode;
    if (!RedisDataUtil.setnx(lockKey, "1", 30))
        throw new IllegalStateException("배포 그룹 수정 중. 잠시 후 재시도.");

    try {
        // CoviMSAPIUtil 기반 ServiceContext 구성 (기존 covimail 인프라 재활용)
        CoviMSAPI api = CoviMSAPIUtil.set(CoviMSAPIEnums.Module.EXCHANGE, param);

        CoviMap result;
        switch (opType) {
            case "CREATE":        result = api.org().addGroup(param);        break;
            case "MODIFY":        result = api.org().modifyGroup(param);     break;
            case "DELETE":        result = api.org().deleteGroup(param);     break;
            case "ADD_MEMBER":
            case "REMOVE_MEMBER": result = api.org().setGroupMember(param);  break;
            default: throw new IllegalArgumentException("Unknown opType: " + opType);
        }
        writeXmlResponse(response, result);  // 기존 XML 응답 방식 재활용
    } finally {
        RedisDataUtil.delete(lockKey);
    }
}
